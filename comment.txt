Arborescence du projet :

backend/
├── controllers/
│   ├── authController.js          # Gère l'authentification des utilisateurs
│   ├── commandeController.js      # Gère les commandes, génération de PDF/Excel, archivage
│   ├── stockController.js         # Gère les stocks (entrées, sorties)
│   ├── adminController.js         # Gère les rôles et permissions des utilisateurs
├── models/
│   ├── userModel.js               # Modèle utilisateur avec rôles
│   ├── commandeModel.js           # Modèle commande avec produits, statut, montant total
│   ├── produitModel.js            # Modèle produit pour gérer les produits en stock
│   ├── stockModel.js              # Modèle stock pour suivre les entrées et sorties de stock
│   ├── journalCommandeModel.js    # Journal des commandes pour suivre les changements de statuts
├── routes/
│   ├── authRoutes.js              # Routes pour l'authentification des utilisateurs
│   ├── commandeRoutes.js          # Routes pour les commandes (création, validation, archivage, rapports)
│   ├── stockRoutes.js             # Routes pour les stocks (entrées, sorties, alertes)
│   ├── adminRoutes.js             # Routes pour la gestion des rôles et permissions
├── services/
│   ├── notificationService.js     # Service pour envoyer les notifications par mail
├── config/
│   └── db.js                      # Configuration de la connexion à MongoDB
├── middlewares/
│   ├── authMiddleware.js          # Middleware d'authentification
│   ├── roleMiddleware.js          # Middleware pour la gestion des rôles des utilisateurs
├── uploads/                       # Dossier pour stocker les fichiers téléchargés (ex : bon de commande Excel)
├── .env                           # Variables d'environnement (configurations sensibles)
├── .gitignore                     # Fichiers et dossiers à ignorer par Git
├── package.json                   # Fichier de configuration des dépendances et scripts Node.js
├── package-lock.json              # Liste des dépendances exactes installées
├── server.js                      # Point d'entrée principal du serveur (démarre l'application)
└── README.md                      # Documentation du projet (à remplir)
